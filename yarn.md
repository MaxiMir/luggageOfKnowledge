# YARN

```shell
yarn init -y # в автоматическом режиме создаст файл манифест package.json

yarn add jquery@1.6 # добавление зависимости

yarn upgrade jquery # обновление зависимости

yarn remove jquery # удаление зависимости

yarn # установка зависимостей (опираясь на информацию из package.json)

yarn list # вывод списка зависимостей. --depth=0 - уровень вложенности

yarn add jquery@3.2.1 --offline # работа оффлайн - установка jquery 3.2.1
```
